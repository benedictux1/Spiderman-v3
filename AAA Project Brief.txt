@app.route('/api/process-note', methods=['POST'])
def process_note_endpoint():
    """Process user-entered notes with AI analysis and RAG context retrieval."""
    try:
        data = request.get_json()
        if not data:
            return jsonify({"error": "No data provided"}), 400
        
        raw_note_text = sanitize_text(data.get('note_text'))
        contact_id = validate_contact_id(data.get('contact_id'))

        if not raw_note_text:
            return jsonify({"error": "Valid note_text is required"}), 400
        if not contact_id:
            return jsonify({"error": "Valid contact_id is required"}), 400

        # --- PRODUCTION RAG PIPELINE ---
        collection_name = f"{ChromaDB.CONTACT_COLLECTION_PREFIX}{contact_id}"
        collection = chroma_client.get_or_create_collection(name=collection_name)
        query_text = " ".join(raw_note_text.split()[:30])
        results = collection.query(query_texts=[query_text], n_results=3)
        retrieved_history = "\n---\n".join(results['documents'][0]) if results['documents'] else "No relevant history found."

        master_prompt = MASTER_PROMPT_TEMPLATE.format(new_note=raw_note_text, history=retrieved_history)
        
        # Debug logging for production monitoring
        print(f"DEBUG: Sending prompt to OpenAI (length: {len(master_prompt)} chars)")
        print(f"DEBUG: First 200 chars: {master_prompt[:200]}...")
        
        try:
            response = openai.ChatCompletion.create(
                model="gpt-4",
                messages=[{"role": "user", "content": master_prompt}],
                max_tokens=DEFAULT_MAX_TOKENS,
                temperature=DEFAULT_AI_TEMPERATURE
            )
            response_content = response.choices[0].message.content.strip()
            print(f"DEBUG: AI Response length: {len(response_content)} chars")
            
        except Exception as openai_error:
            print(f"DEBUG: OpenAI API Error: {openai_error}")
            return jsonify({"error": f"OpenAI API Error: {str(openai_error)}"}), 500
        
        # Production JSON parsing with multiple fallback strategies
        if response_content.startswith('```json'):
            response_content = response_content.replace('```json', '').replace('```', '').strip()
        elif response_content.startswith('```'):
            response_content = response_content.replace('```', '').strip()
        
        try:
            ai_json_response = json.loads(response_content)
            print(f"DEBUG: JSON parsing successful. Keys: {list(ai_json_response.keys())}")
        except json.JSONDecodeError as e:
            print(f"DEBUG: JSON parsing failed: {e}")
            # Fallback regex extraction for malformed JSON
            import re
            json_match = re.search(r'\{.*\}', response_content, re.DOTALL)
            if json_match:
                try:
                    ai_json_response = json.loads(json_match.group())
                    print(f"DEBUG: Regex extraction successful. Keys: {list(ai_json_response.keys())}")
                except Exception as regex_error:
                    print(f"DEBUG: Regex extraction failed: {regex_error}")
                    return jsonify({"error": f"Failed to parse AI response: {response_content[:200]}..."}), 500
            else:
                print("DEBUG: No JSON found in response")
                return jsonify({"error": f"Failed to parse AI response: {response_content[:200]}..."}), 500
        
        return jsonify(ai_json_response)
    except Exception as e:
        return jsonify({"error": f"An internal error occurred: {e}"}), 500
```

def run_telegram_import(task_id, identifier, contact_id, days_back):
    """Sync wrapper for async Telegram import with progress tracking."""
    try:
        loop = asyncio.get_running_loop()
    except RuntimeError:
        loop = asyncio.new_event_loop()
        asyncio.set_event_loop(loop)

    try:
        loop.run_until_complete(
            _async_run_telegram_import(task_id, identifier, contact_id, days_back)
        )
    except Exception as e:
        update_task_status(task_id, "failed", "Critical error in async runner.", str(e), progress=100)

async def _async_run_telegram_import(task_id, identifier, contact_id, days_back):
    """Main async import function with comprehensive error handling."""
    update_task_status(task_id, "connecting", "Initializing import...", progress=5)

    client = TelegramClient(SESSION_NAME, API_ID, API_HASH)
    try:
        update_task_status(task_id, "connecting", "Connecting to Telegram...", progress=10)
        await asyncio.wait_for(client.connect(), timeout=45.0)

        if not await client.is_user_authorized():
            update_task_status(task_id, "failed", "Telegram authentication required.", 
                               "Session not authorized. Run interactive setup first.", progress=100)
            return

        update_task_status(task_id, "fetching", f"Finding contact '{identifier}'...", progress=20)
        entity = await client.get_entity(identifier)

        # Check last sync for incremental updates
        conn = get_db_connection()
        try:
            cursor = conn.execute('SELECT telegram_last_sync FROM contacts WHERE id = ?', (contact_id,))
            result = cursor.fetchone()
            last_sync_timestamp = result['telegram_last_sync'] if result and result['telegram_last_sync'] else None
        finally:
            conn.close()
        
        if last_sync_timestamp:
            since_date = datetime.fromisoformat(last_sync_timestamp)
            status_msg = f"Fetching new messages since last sync: {since_date.strftime('%Y-%m-%d %H:%M')}..."
        else:
            since_date = datetime.now() - timedelta(days=days_back)
            status_msg = f"Fetching messages from last {days_back} days..."
        update_task_status(task_id, "fetching", status_msg, progress=30)

        transcript = []
        message_count = 0
        async for message in client.iter_messages(entity, offset_date=since_date, reverse=True, limit=2000):
            if message.text:
                sender_name = "Me" if message.out else (entity.first_name or entity.username or str(entity.id))
                formatted_message = f"[{message.date.strftime('%Y-%m-%d %H:%M')}] {sender_name}: {message.text}"
                transcript.append(formatted_message)
                message_count += 1
                if message_count % 20 == 0:
                    update_task_status(task_id, "fetching", f"Fetched {message_count} messages...", progress=min(80, 30 + (message_count // 50)))

        if not transcript:
            update_task_status(task_id, "completed", "No new messages found.", progress=100)
            return

        full_transcript = "\n".join(transcript)
        update_task_status(task_id, "processing", f"Sending {message_count} messages for AI analysis...", progress=85)

        headers = {'Authorization': f'Bearer {KITH_API_TOKEN}', 'Content-Type': 'application/json'}
        payload = {'contact_id': contact_id, 'transcript': full_transcript}
        
        response = requests.post(f"{KITH_API_URL}/api/process-transcript", headers=headers, json=payload, timeout=300)

        if response.status_code == 200:
            # Update last sync timestamp
            conn = get_db_connection()
            try:
                conn.execute('UPDATE contacts SET telegram_last_sync = ? WHERE id = ?', (datetime.now().isoformat(), contact_id))
                conn.commit()
            finally:
                conn.close()
            update_task_status(task_id, "completed", f"Successfully imported and analyzed {message_count} messages.", progress=100)
        else:
            error_detail = response.text
            try:
                error_detail = response.json().get('error', response.text)
            except:
                pass
            update_task_status(task_id, "failed", f"API processing failed: HTTP {response.status_code}", error_detail, progress=100)

    except asyncio.TimeoutError:
        update_task_status(task_id, "failed", "Connection to Telegram timed out.", "Could not connect to Telegram servers within 45 seconds.", progress=100)
    except ValueError as e:
        update_task_status(task_id, "failed", f"Could not find user '{identifier}'", f"Telegram entity not found: {str(e)}", progress=100)
    except Exception as e:
        update_task_status(task_id, "failed", "An unexpected error occurred during import.", str(e), progress=100)
    finally:
        if client.is_connected():
            await client.disconnect()
```

def update_task_status(task_id, status, message="", error="", progress=None):
    """Thread-safe task status updates with retry logic."""
    max_retries = 3
    retry_delay = 1
    
    for attempt in range(max_retries):
        try:
            conn = get_db_connection()
            try:
                # Verify task exists
                cursor = conn.execute('SELECT id FROM import_tasks WHERE id = ?', (task_id,))
                if not cursor.fetchone():
                    print(f"Warning: Task {task_id} not found in database")
                    return
                
                # Update with progress tracking
                update_params = [status, message[:1000] if message else "", error[:2000] if error else ""]
                
                if progress is not None:
                    progress = max(0, min(100, int(progress)))
                    update_params.append(progress)
                    update_params.append(task_id)
                    
                    conn.execute('''
                        UPDATE import_tasks 
                        SET status = ?, status_message = ?, error_details = ?, progress = ?
                        WHERE id = ?
                    ''', update_params)
                else:
                    update_params.append(task_id)
                    conn.execute('''
                        UPDATE import_tasks 
                        SET status = ?, status_message = ?, error_details = ?
                        WHERE id = ?
                    ''', update_params)
                
                # Set completion timestamp for final states
                if status in ['completed', 'failed']:
                    conn.execute('''
                        UPDATE import_tasks 
                        SET completed_at = CURRENT_TIMESTAMP
                        WHERE id = ?
                    ''', (task_id,))
                
                conn.commit()
                print(f"[Task {task_id}] Status: {status} ({progress or ''}%) - {message}")
                return
                
            finally:
                conn.close()
                
        except Exception as e:
            if attempt < max_retries - 1:
                print(f"Error updating task status (attempt {attempt + 1}/{max_retries}): {e}")
                time.sleep(retry_delay)
                retry_delay *= 2
            else:
                print(f"Failed to update task status after {max_retries} attempts: {e}")
```

@app.route('/api/import-vcard', methods=['POST'])
def import_vcard_endpoint():
    """Parses an uploaded .vcf file and creates contacts, avoiding duplicates."""
    if 'vcard_file' not in request.files:
        return jsonify({"error": "No file part"}), 400

    file = request.files['vcard_file']
    if file.filename == '' or not file.filename.endswith('.vcf'):
        return jsonify({"error": "No selected or invalid file type"}), 400

    try:
        vcf_data = file.read().decode('utf-8')
        contacts_created = 0
        contacts_skipped = 0
        session = get_session()

        # Parse VCard data with comprehensive error handling
        vcards = list(vobject.readComponents(vcf_data))
        
        for vcard in vcards:
            if hasattr(vcard, 'fn') and vcard.fn.value:
                full_name = vcard.fn.value.strip()

                # Enhanced duplicate detection (case-insensitive)
                existing_contact = session.query(Contact).filter(
                    Contact.full_name.ilike(full_name),
                    Contact.user_id == 1
                ).first()

                if existing_contact:
                    contacts_skipped += 1
                    continue

                # Create new contact with proper field initialization
                new_contact = Contact(
                    full_name=full_name,

def search_endpoint():
    """Unified search across contacts and their data."""
    query = request.args.get('q')
    if not query or len(query) < 2:
        return jsonify([])

    try:
        conn = get_db_connection()
        try:
            # 1. Keyword Search using direct SQL for performance
            cursor = conn.execute('''
                SELECT DISTINCT c.id 
                FROM contacts c 
                LEFT JOIN synthesized_entries se ON c.id = se.contact_id 
                WHERE (c.full_name LIKE ? OR se.content LIKE ?) AND c.user_id = 1
            ''', (f'%{query}%', f'%{query}%'))
            
            keyword_contact_ids = [row['id'] for row in cursor.fetchall()]

            # 2. Semantic Search (ChromaDB Vector Database)
            try:
                master_collection = chroma_client.get_or_create_collection(name=ChromaDB.MASTER_COLLECTION_NAME)
                semantic_results = master_collection.query(query_texts=[query], n_results=10)
                semantic_contact_ids = [int(meta['contact_id']) for meta in semantic_results['metadatas'][0]]
            except Exception:
                semantic_contact_ids = []

            # 3. Combine and De-duplicate Results
            combined_ids = list(set(keyword_contact_ids + semantic_contact_ids))

            # 4. Fetch Contact Details for matched IDs
            if not combined_ids:
                return jsonify([])

            # Convert to SQL IN clause for performance
            placeholders = ','.join('?' * len(combined_ids))
            cursor = conn.execute(f'''
                SELECT id, full_name, tier 
                FROM contacts 
                WHERE id IN ({placeholders}) AND user_id = 1
            ''', combined_ids)
            
            final_contacts = [dict(row) for row in cursor.fetchall()]
            
            return jsonify(final_contacts)
            
        finally:
            conn.close()

    except Exception as e:
        logger.error(f"Search failed: {e}")

class RelationshipAnalytics:
    """Advanced analytics for relationship health and insights."""
    
    def calculate_relationship_health_score(self, contact_id: int) -> Dict:
        """Calculate comprehensive relationship health score using multi-factor algorithm."""
        with self.get_connection() as conn:
            cursor = conn.cursor()
            
            # Get all approved entries for this contact
            cursor.execute("""
                SELECT category, ai_confidence, created_at, is_approved
                FROM synthesized_entries 
                WHERE contact_id = ? AND is_approved = TRUE
                ORDER BY created_at DESC
            """, (contact_id,))
            
            entries = cursor.fetchall()
            
            if not entries:
                return {
                    "health_score": 0,
                    "total_interactions": 0,
                    "last_interaction": None,
                    "category_distribution": {},
                    "confidence_avg": 0,
                    "insights": ["No data available for this contact"]
                }
            
            # Calculate core metrics
            total_interactions = len(entries)
            confidence_scores = [entry[1] for entry in entries if entry[1] is not None]
            avg_confidence = sum(confidence_scores) / len(confidence_scores) if confidence_scores else 0
            
            # Category distribution analysis
            categories = [entry[0] for entry in entries]
            category_dist = Counter(categories)
            
            # RECENCY SCORE: More recent interactions = better relationship health
            latest_date = max(entry[2] for entry in entries)
            days_since_last = (datetime.now() - datetime.fromisoformat(latest_date)).days
            recency_score = max(0, 100 - (days_since_last * 2))  # Lose 2 points per day
            
            # ENGAGEMENT SCORE: Higher interaction frequency = stronger relationship
            if total_interactions > 0:
                first_date = min(entry[2] for entry in entries)
                total_days = (datetime.fromisoformat(latest_date) - datetime.fromisoformat(first_date)).days + 1
                interactions_per_week = (total_interactions / total_days) * 7
                engagement_score = min(100, interactions_per_week * 10)  # 10 interactions/week = 100 score
            else:
                engagement_score = 0
            
            # QUALITY SCORE: Higher AI confidence = better data quality
            quality_score = min(100, avg_confidence * 10)  # Convert 0-10 scale to 0-100
            
            # DIVERSITY SCORE: More categories = better relationship understanding
            diversity_score = min(100, len(category_dist) * 5)  # 20 categories = 100 score
            
            # WEIGHTED HEALTH SCORE CALCULATION
            health_score = (recency_score * 0.3 +      # 30% weight on recency
                          engagement_score * 0.3 +   # 30% weight on engagement
                          quality_score * 0.2 +      # 20% weight on data quality
                          diversity_score * 0.2)       # 20% weight on diversity
            
            # Generate insights based on scores
            insights = self._generate_insights(entries, health_score)
            
            return {
                "health_score": round(health_score, 1),
                "total_interactions": total_interactions,
                "last_interaction": latest_date,
                "category_distribution": dict(category_dist),
                "confidence_avg": round(avg_confidence, 2),
                "recency_score": round(recency_score, 1),
                "engagement_score": round(engagement_score, 1),
                "quality_score": round(quality_score, 1),
                "diversity_score": round(diversity_score, 1),
                "insights": insights
            }

    def extract_date_time_from_text(self, text: str) -> Optional[Dict]:

        # Comprehensive regex patterns for date/time extraction
        patterns = [

            # "next week on Monday"
            r'next\s+week\s+(?:on\s+)?(monday|tuesday|wednesday|thursday|friday|saturday|sunday)',

            r'next\s+month\s+(?:on\s+the\s+)?(\d{1,2})(?:st|nd|rd|th)?',
            # "on December 25th"
            r'on\s+(january|february|march|april|may|june|july|august|september|october|november|december)\s+(\d{1,2})(?:st|nd|rd|th)?',
            # "at 2pm on Friday"
            r'at\s+(\d{1,2})(?::(\d{2}))?\s*(am|pm)\s+(?:on\s+)?(monday|tuesday|wednesday|thursday|friday|saturday|sunday)',

            r'this\s+(monday|tuesday|wednesday|thursday|friday|saturday|sunday)',
            # "next Friday"
            r'next\s+(monday|tuesday|wednesday|thursday|friday|saturday|sunday)',
        ]
        
        text_lower = text.lower()
        
        for pattern in patterns:
            match = re.search(pattern, text_lower)
            if match:
                return self._parse_date_time_match(match, text_lower)
        
        return None
    
    def _parse_date_time_match(self, match, text: str) -> Dict:
        """Parse matched date/time information into structured format."""
        now = datetime.now()
        
        if 'tomorrow' in text:
            target_date = now + timedelta(days=1)
            if match.group(1):  # Has time
                hour = int(match.group(1))
                minute = int(match.group(2)) if match.group(2) else 0
                am_pm = match.group(3)
                
                if am_pm == 'pm' and hour != 12:
                    hour += 12
                elif am_pm == 'am' and hour == 12:
                    hour = 0
                
                target_date = target_date.replace(hour=hour, minute=minute, second=0, microsecond=0)
            
            return {
                'datetime': target_date.isoformat(),
                'confidence': 0.9,
                'extracted_from': text,
                'type': 'specific_datetime' if match.group(1) else 'date_only'
            }
        
        # Similar logic for other patterns...
        # (Additional parsing logic for different date patterns)
        
        return {
            'datetime': now.isoformat(),
            'confidence': 0.5,
            'extracted_from': text,
            'type': 'uncertain'
        }
    
    def create_event_from_actionable_item(self, contact_name: str, summary: str, date_time_info: Dict) -> Dict:
        """Create calendar event from actionable item with extracted date/time."""
        event = {
            'id': str(uuid.uuid4()),
            'title': f"Follow up with {contact_name}",
            'description': summary,
            'datetime': date_time_info['datetime'],
            'duration_minutes': 30,
            'confidence': date_time_info['confidence'],
            'source': 'kith_actionable_item',
            'contact_name': contact_name,
            'created_at': datetime.now().isoformat()
        }
        
        return event
    
    def save_events_to_local_calendar(self, events: List[Dict]):
        """Save events to local JSON calendar file (demo implementation)."""
        calendar_file = 'kith_calendar_events.json'
        
        try:
            # Load existing events
            if os.path.exists(calendar_file):
                with open(calendar_file, 'r') as f:
                    existing_events = json.load(f)
            else:
                existing_events = []
            
            # Add new events
            existing_events.extend(events)
            
            # Save updated events
            with open(calendar_file, 'w') as f:
                json.dump(existing_events, f, indent=2)
            
            return True
            
        except Exception as e:
            print(f"Error saving events: {e}")
            return False
```

@app.route('/api/contacts', methods=['POST'])
def create_contact():
    """Create a new contact manually."""
    try:
        data = request.get_json()
        full_name = data.get('full_name')
        tier = data.get('tier', 2)
        
        if not full_name or not full_name.strip():
            return jsonify({"error": "Full name is required"}), 400
        
        session = get_session()
        
        # Check for existing contact by full name (case-insensitive)
        existing_contact = session.query(Contact).filter(
            Contact.full_name.ilike(full_name.strip()),
            Contact.user_id == 1
        ).first()
        
        if existing_contact:
            return jsonify({"error": f"Contact '{full_name}' already exists"}), 409
        
        # Create new contact
        new_contact = Contact(
            full_name=full_name.strip(),
            tier=tier,
            user_id=1,  # Assume user_id=1 for MVP
            vector_collection_id=f"contact_{uuid.uuid4().hex[:8]}"
        )
        
        session.add(new_contact)
        session.commit()
        
        return jsonify({
            "id": new_contact.id,
            "full_name": new_contact.full_name,
            "tier": new_contact.tier,
            "created_at": new_contact.created_at.isoformat()
        }), 201
        
    except Exception as e:
        session.rollback()
        return jsonify({"error": f"Failed to create contact: {e}"}), 500
    finally:
        session.close()
```

##### 2.2 Advanced Entity Resolution System

**Multi-Format Identifier Processing:**
```python
class IdentifierResolver:
    """
    Advanced entity resolution system supporting multiple identifier formats
    with fuzzy matching and validation capabilities.
    """
    
    IDENTIFIER_PATTERNS = {
        'phone': r'^\+?[1-9]\d{1,14}$',  # E.164 format validation
        'username': r'^@?[a-zA-Z][\w]{4,31}$',  # Telegram username format
        'user_id': r'^\d{1,10}$',  # Numeric user ID
        'display_name': r'^.{1,64}$'  # Any display name up to 64 chars
    }
    
    async def resolve_entity(self, client, identifier):
        """
        Resolve identifier to Telegram entity with comprehensive fallback logic.
        
        Resolution Pipeline:
        1. Direct username lookup (@username)
        2. Phone number resolution (+1234567890)
        3. User ID lookup (numeric ID)
        4. Display name fuzzy search
        5. Partial name matching across dialog history
        """
        
        # Step 1: Normalize and categorize identifier
        normalized_id, id_type = self._normalize_identifier(identifier)
        
        # Step 2: Direct resolution attempts
        if id_type == 'username':
            entity = await self._resolve_by_username(client, normalized_id)
            if entity:
                return entity, 'username_direct'
                
        elif id_type == 'phone':
            entity = await self._resolve_by_phone(client, normalized_id)
            if entity:
                return entity, 'phone_direct'
                
        elif id_type == 'user_id':
            entity = await self._resolve_by_user_id(client, normalized_id)
            if entity:
                return entity, 'user_id_direct'
        
        # Step 3: Fallback to dialog search with fuzzy matching
        return await self._resolve_by_dialog_search(client, identifier)
    
    async def _resolve_by_dialog_search(self, client, identifier):
        """
        Search through user's dialog list for matching entities.
        Implements sophisticated matching algorithms including:
        - Exact display name match
        - Partial display name match (minimum 3 characters)
        - Levenshtein distance matching (max distance: 2)
        - Username partial matching
        - Phone number partial matching
        """
        
        matches = []
        search_term = identifier.lower().strip('@')
        
        async for dialog in client.iter_dialogs(limit=500):  # Reasonable limit for performance
            entity = dialog.entity
            
            if not isinstance(entity, (User, Chat, Channel)):
                continue
                
            # Score potential matches
            match_score = self._calculate_match_score(entity, search_term)
            
            if match_score >= self.MINIMUM_MATCH_THRESHOLD:
                matches.append((entity, match_score, self._get_match_reason(entity, search_term)))
        
        # Return best match if confidence is high enough
        if matches:
            best_match = max(matches, key=lambda x: x[1])
            if best_match[1] >= self.HIGH_CONFIDENCE_THRESHOLD:
                return best_match[0], f'fuzzy_match_{best_match[2]}'
        
        return None, 'no_match_found'
```

class ConversationImporter:

    Sophisticated conversation import engine with advanced filtering,
    deduplication, and content processing capabilities.
    """
    
    def __init__(self, client, entity, parameters):
        self.client = client
        self.entity = entity
        self.params = parameters

        self.media_downloader = MediaDownloader() if parameters.get('include_media') else None
        self.deduplication_cache = set()
        
    async def import_conversations(self):
        """
        Main import pipeline with sophisticated message filtering and processing.
        """
        
        # Configure import parameters
        end_date = datetime.now()
        start_date = end_date - timedelta(days=self.params['days_back'])
        message_limit = self.params.get('message_limit', 1000)
        
        # Message collection with real-time processing
        collected_messages = []
        message_count = 0
        
        async for message in self.client.iter_messages(
            self.entity, 
            limit=message_limit,
            offset_date=end_date,
            reverse=True  # Chronological order for better context
        ):
            
            # Skip messages outside date range
            if message.date < start_date:
                break
                
            # Skip deleted or invalid messages
            if not self._is_valid_message(message):
                continue
                
            # Deduplication check
            message_hash = self._generate_message_hash(message)
            if message_hash in self.deduplication_cache:
                continue
                
            self.deduplication_cache.add(message_hash)
            
            # Process message content
            processed_message = await self._process_message(message)
            collected_messages.append(processed_message)
            
            message_count += 1
            
            # Progress reporting
            if message_count % 50 == 0:
                await self._report_progress(message_count, message_limit)
        
        return self._prepare_conversation_data(collected_messages)
    
    async def _process_message(self, message):
        """
        Comprehensive message processing including content extraction,
        media handling, and metadata enrichment.
        """
        
        processed_data = {
            'id': message.id,
            'date': message.date.isoformat(),
            'from_id': getattr(message.from_id, 'user_id', None) if message.from_id else None,
            'text': message.text or '',
            'media_type': None,
            'media_url': None,
            'reply_to': message.reply_to_msg_id,
            'forward_info': self._extract_forward_info(message),
            'edit_date': message.edit_date.isoformat() if message.edit_date else None,
            'reactions': self._extract_reactions(message),
            'sentiment': None,  # Will be filled by AI analysis
            'entities': self._extract_text_entities(message)
        }
        
        # Media processing
        if message.media and self.media_downloader:
            media_info = await self._process_media(message)
            processed_data.update(media_info)
        
        # Message type classification
        processed_data['message_type'] = self._classify_message_type(message)
        
        return processed_data

```python
class TelegramSecurityManager:
    """
    Comprehensive security manager for Telegram integration operations.
    Implements defense-in-depth security principles.
    """
    
    def __init__(self):
        self.rate_limiter = RateLimiter()
        self.access_logger = SecurityLogger()
        self.encryption_manager = EncryptionManager()
        
    async def validate_import_request(self, user_id, request_data):
        """
        Multi-layer security validation for import requests.
        """
        
        # 1. Rate limiting check
        if not self.rate_limiter.is_allowed(user_id, 'telegram_import'):
            raise RateLimitExceededError("Import rate limit exceeded. Please wait before trying again.")
        
        # 2. User permission validation
        user_permissions = await self.get_user_permissions(user_id)
        if not user_permissions.can_import_telegram:
            raise PermissionDeniedError("User does not have Telegram import permissions.")
        
        # 3. Input sanitization and validation
        sanitized_data = self.sanitize_import_request(request_data)
        
        # 4. Resource usage validation
        if sanitized_data.get('message_limit', 0) > user_permissions.max_message_limit:
            raise ResourceLimitExceededError(f"Message limit exceeds maximum allowed ({user_permissions.max_message_limit})")
        
        # 5. Identifier validation for security
        self.validate_identifier_security(sanitized_data.get('identifier'))
        
        return sanitized_data
    
    def validate_identifier_security(self, identifier):
        """
        Security validation for user-provided identifiers.
        Prevents injection attacks and malicious input.
        """
        
        if not identifier or len(identifier) > 255:
            raise ValidationError("Invalid identifier length")
        
        # Check for potential injection patterns
        dangerous_patterns = [
            r'[<>"\']',  # HTML/XML injection
            r'[;\|\&\$]',  # Command injection
            r'(union|select|drop|delete|insert|update)\s',  # SQL injection
            r'javascript:',  # XSS
            r'data:',  # Data URI
        ]
        
        for pattern in dangerous_patterns:
            if re.search(pattern, identifier, re.IGNORECASE):
                raise SecurityValidationError("Identifier contains potentially malicious content")
        
        # Normalize and validate format
        normalized = identifier.strip().lower()
        
        # Additional security checks based on identifier type
        if normalized.startswith('+'):
            # Phone number security validation
            self._validate_phone_number_security(normalized)
        elif normalized.startswith('@'):
            # Username security validation
            self._validate_username_security(normalized)
        else:
            # Display name security validation
            self._validate_display_name_security(normalized)

class EncryptionManager:
    """
    Handle encryption/decryption of sensitive data including credentials and messages.
    """
    
    def __init__(self):
        self.key = self._load_encryption_key()
        self.cipher = Fernet(self.key)
        
    def encrypt_credentials(self, api_id, api_hash):
        """Encrypt Telegram API credentials for secure storage."""
        credentials = {
            'api_id': api_id,
            'api_hash': api_hash,
            'encrypted_at': datetime.utcnow().isoformat()
        }
        
        serialized = json.dumps(credentials).encode()
        encrypted = self.cipher.encrypt(serialized)
        
        return base64.b64encode(encrypted).decode()
    
    def decrypt_credentials(self, encrypted_credentials):
        """Decrypt Telegram API credentials for use."""
        try:
            decoded = base64.b64decode(encrypted_credentials.encode())
            decrypted = self.cipher.decrypt(decoded)
            credentials = json.loads(decrypted.decode())
            
            # Validate credential freshness (expire after 30 days)
            encrypted_at = datetime.fromisoformat(credentials['encrypted_at'])
            if (datetime.utcnow() - encrypted_at).days > 30:
                raise CredentialExpiredError("Encrypted credentials have expired")
            
            return credentials['api_id'], credentials['api_hash']
            
        except Exception as e:
            raise DecryptionError(f"Failed to decrypt credentials: {str(e)}")

        if 'unauthorized' in error_message or 'auth' in error_message:
            return self._create_error_info('authentication', error)
        elif 'rate limit' in error_message or 'flood' in error_message:
            return self._create_error_info('rate_limit', error)
        elif 'network' in error_message or 'connection' in error_message:
            return self._create_error_info('network', error)
        elif 'not found' in error_message or 'no such' in error_message:
            return self._create_error_info('entity_not_found', error)

            return self._create_error_info('permission_denied', error)

        else:

        """Execute appropriate recovery strategy based on error type."""

            delay = min(300, 5 * (2 ** context.get('retry_count', 0)))  # Max 5 minutes
            await asyncio.sleep(delay)
            return {'action': 'retry', 'delay': delay}

            delay = min(60, 2 ** context.get('retry_count', 0))  # Max 1 minute
            await asyncio.sleep(delay)
            return {'action': 'retry', 'delay': delay}
            
        elif recovery_action == 'suggest_alternatives':
            alternatives = await self._suggest_identifier_alternatives(context.get('identifier'))
            return {'action': 'suggest_alternatives', 'alternatives': alternatives}
            
        else:
            return {'action': 'abort', 'reason': recovery_action}

class ImportTaskRecovery:
    """
    Handle recovery of interrupted or failed import tasks.
    """
    
    async def recover_failed_task(self, task_id):
        """
        Attempt to recover a failed import task by resuming from last checkpoint.
        """
        
        task_record = await self._get_task_record(task_id)
        
        if not task_record:
            raise TaskNotFoundError(f"Task {task_id} not found")
        
        if task_record.status not in ['failed', 'interrupted']:
            raise InvalidTaskStateError(f"Task {task_id} is not in recoverable state")
        
        # Determine recovery point
        last_checkpoint = await self._get_last_checkpoint(task_id)
        
        if last_checkpoint:
            # Resume from checkpoint
            return await self._resume_from_checkpoint(task_record, last_checkpoint)
        else:
            # Restart from beginning
            return await self._restart_task(task_record)
    
    async def _create_recovery_checkpoint(self, task_id, progress_data):
        """Create checkpoint for task recovery."""
        
        checkpoint_data = {
            'task_id': task_id,
            'progress': progress_data['progress'],
            'last_message_id': progress_data.get('last_message_id'),
            'processed_count': progress_data.get('processed_count', 0),
            'created_at': datetime.utcnow()
        }
        
        await self._save_checkpoint(checkpoint_data)

Kith Telegram Integration - Technical Implementation Brief (V3)

Version: 3.0

Date: August 9, 2025

Status: Definitive Developer Blueprint for Telegram Feature

Author: Gemini

1. Executive Overview & Architectural Rationale

This document details the complete re-architecture of the Kith platform's Telegram integration. The previous implementation, a standalone script with direct database access, has proven to be fragile, leading to schema mismatch errors and a clunky user workflow.

The new architecture is a decoupled, asynchronous background task system. This design solves the core issues by establishing the main Flask application as the single source of truth for all database operations and AI analysis. The Telegram-specific logic is isolated into a background worker, which communicates with the main application via secure API endpoints.

Key Benefits of This New Architecture:

Eliminates Schema Errors: The worker never writes to the database schema directly, completely removing the risk of no such column or NOT NULL constraint errors.

Robust & Maintainable: Logic is cleanly separated. The Flask app handles web requests and data persistence, while the worker handles the long-running task of communicating with Telegram.

Superior User Experience: The import process is initiated from the web UI and runs in the background. The user receives real-time progress updates without needing to manage a command-line script.

Reliable API Communication: The worker calls the running Flask server's API, eliminating the "Connection refused" errors.

This brief provides all necessary schemas, code snippets, and logic for a developer to implement this robust solution.

2. High-Level Architectural Flow

The new workflow follows a clear, decoupled pattern:

UI (Initiation): The user enters a Telegram identifier (e.g., @CrisRed) and clicks "Start Import" on the Kith web app's settings page.

Flask API (Job Creation): The frontend calls a new /api/telegram/start-import endpoint. This endpoint's only job is to create a task record in the database with a pending status and add a job to a background task queue. It immediately returns a task_id to the UI.

UI (Polling): The frontend uses the task_id to periodically call a /api/telegram/import-status/<task_id> endpoint to get real-time progress updates.

Background Worker (Execution): A separate worker process, managed by a scheduler, picks up the job from the queue.

It updates the task status in the database (e.g., to connecting).

It securely connects to Telegram using the Telethon library.

It fetches all new messages since the last successful import for that contact.

It formats the messages into a clean transcript.

Flask API (Processing & Saving): The worker makes an authenticated POST request to the existing /api/process-transcript endpoint, sending the transcript. The main Flask app then performs the AI analysis and handles all database writes, ensuring consistency.

UI (Completion): The worker updates the task's final status to completed or failed. The UI polling reflects this, informing the user that the import is finished.

import os

def update_task_status(task_id, status, message="", error=""):
    """Helper function to update the task status in the database."""

def run_telegram_import(task_id, identifier, contact_id, days_back):
    """The main background job function."""
    
    update_task_status(task_id, "connecting", "Connecting to Telegram...")
    
    try:
        with TelegramClient(SESSION_NAME, API_ID, API_HASH) as client:
            update_task_status(task_id, "fetching", f"Finding contact '{identifier}'...")
            
            entity = client.get_entity(identifier)
            
            # --- Database Logic ---
            # last_sync_timestamp = db.get_last_sync_for_contact(contact_id)
            last_sync_timestamp = None # Placeholder
            
            if last_sync_timestamp:
                since_date = last_sync_timestamp
            else:
                since_date = datetime.now() - timedelta(days=days_back)

            update_task_status(task_id, "fetching", f"Fetching new messages since {since_date.strftime('%Y-%m-%d')}...")

            transcript = []
            messages = client.iter_messages(entity, offset_date=since_date, reverse=True)
            
            for message in messages:
                if message.text:
                    sender_name = "Me" if message.out else entity.first_name
                    formatted_message = f"[{message.date.strftime('%Y-%m-%d %H:%M')}] {sender_name}: {message.text}"
                    transcript.append(formatted_message)
            
            if not transcript:
                update_task_status(task_id, "completed", "No new messages found.")
                return

            full_transcript = "\n".join(transcript)
            
            update_task_status(task_id, "processing", "Sending transcript for AI analysis...")

            # --- API Call to Main App ---
            headers = {
                'Authorization': f'Bearer {KITH_API_TOKEN}',
                'Content-Type': 'application/json'
            }
            payload = {
                'contact_id': contact_id,
                'transcript': full_transcript
            }
            
            response = requests.post(f"{KITH_API_URL}/api/process-transcript", headers=headers, json=payload)
            
            if response.status_code == 200:
                # --- Database Logic ---
                # db.update_last_sync_for_contact(contact_id, datetime.now())
                update_task_status(task_id, "completed", "Analysis complete and data saved.")
            else:
                raise Exception(f"API processing failed: {response.text}")

    except Exception as e:
        update_task_status(task_id, "failed", "An error occurred.", error=str(e))

function pollTaskStatus(taskId) {
    const intervalId = setInterval(async () => {
        try {
            const response = await fetch(`/api/telegram/import-status/${taskId}`);
            const data = await response.json();

            statusArea.innerHTML = `
                <p><strong>Task ID:</strong> ${data.task_id}</p>
                <p><strong>Status:</strong> ${data.status}</p>
                <p>${data.status_message || ''}</p>
            `;

            if (data.status === 'completed' || data.status === 'failed') {
                clearInterval(intervalId);
                startImportBtn.disabled = false;
                if (data.error_details) {
                    statusArea.innerHTML += `<p style="color:red;"><strong>Error:</strong> ${data.error_details}</p>`;
                }
            }
        } catch (error) {
            clearInterval(intervalId);
            statusArea.innerHTML = `<p style="color:red;">Failed to poll task status.</p>`;
            startImportBtn.disabled = false;
        }
    }, 3000); // Poll every 3 seconds
}

// Add to the main <script> tag
const rawLogsDetails = document.getElementById('raw-logs-details');
const rawLogsContent = document.getElementById('raw-logs-content');

rawLogsDetails.addEventListener('toggle', async (event) => {
    // Only fetch data if the section is opened and hasn't been loaded yet
    if (event.target.open && rawLogsContent.children.length === 0) {
        rawLogsContent.innerHTML = '<p>Loading history...</p>';
        try {
            const response = await fetch(`/api/contact/${currentProfileContactId}/raw-logs`);
            const logs = await response.json();

            if (logs.length === 0) {
                rawLogsContent.innerHTML = '<p>No raw notes have been saved for this contact yet.</p>';
                return;
            }

            let logsHtml = '<ul>';
            logs.forEach(log => {
                const formattedDate = new Date(log.date).toLocaleString();
                logsHtml += `<li><strong>${formattedDate}:</strong><pre>${log.content}</pre></li>`;
            });
            logsHtml += '</ul>';

            rawLogsContent.innerHTML = logsHtml;

        } catch (error) {
            rawLogsContent.innerHTML = '<p style="color:red;">Failed to load raw history.</p>';
        }
    }
});
